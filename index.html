Code sample - PDO scorecard mapping

# Maps calibrated P(good) → interpretable scorecard score # PDO = 20 pts to double the odds | SCORE0 = 600 at ODDS0 = 50 import numpy as np def prob_to_score(p_good, score0=600, odds0=50, pdo=20): odds = np.array(p_good) / (1 - np.array(p_good) + 1e-9) factor = pdo / np.log(2) offset = score0 - factor * np.log(odds0) return np.round(offset + factor * np.log(odds)).astype(int) # P(good) = 0.09 → score ≈ 421 → DECLINE # P(good) = 0.86 → score ≈ 650 → APPROVE
